\textcolor{family-vaes}{\family{family-vaes}{Variational Autoencoders (VAEs)}:}
\begin{itemize}
  \item Vanilla VAE
  \item $\beta$-VAE
  \item VQ-VAE
  \item VAE-GAN
  \item CVAE (Conditional VAE)
  \item DFC-VAE (Disentangled Feature Control VAE)
  \item HiVAE (Hierarchical VAE)
  \item VLAE (Variational Lossy Autoencoder)
  \item AdaVAE (Adaptive VAE)
  \item SCVAE (Semi-Supervised Conditional VAE)
  \item AAE (Adversarial Autoencoder)
\end{itemize}

\textcolor{family-gans}{\family{family-gans}{Generative Adversarial Networks (GANs)}:}
\begin{itemize}
  \item Vanilla GAN
  \item DCGAN (Deep Convolutional GAN)
  \item CGAN (Conditional GAN)
  \item WGAN (Wasserstein GAN)
  \item LSGAN (Least Squares GAN)
  \item CycleGAN
  \item StyleGAN
  \item ProGAN (Progressive GAN)
  \item BigGAN
  \item SNGAN (Spectral Normalization GAN)
  \item GLO (Generative Latent Optimization)
  \item MSG-GAN (Multi-Scale Gradient GAN)
  \item RGAN (Relativistic GAN)
  \item BiGAN (Bidirectional GAN)
  \item TripleGAN
  \item MoCoGAN (Motion Conditioned GAN)
  \item StarGAN
  \item DALL-E (GAN-based image generation model by OpenAI)
  \item Swapping Autoencoder GAN (SWAGAN)
\end{itemize}

\textcolor{family-flows}{\family{family-flows}{Flow-based Models}:}
\begin{itemize}
  \item RealNVP (Real-valued Non-Volume Preserving)
  \item Glow (Generative Flow with Invertible 1x1 Convolutions)
  \item FFJORD (Free-form Jacobian Adversarially Regularized Dynamics)
  \item CNFs (Continuous Normalizing Flows)
  \item MAF (Masked Autoregressive Flow)
  \item IAF (Inverse Autoregressive Flow)
  \item Neural Spline Flows
  \item DAF (Denoising Autoencoder Flow)
  \item AugVAE (Augmented VAE)
\end{itemize}

\textcolor{family-autoregressive}{\family{family-autoregressive}{Auto-regressive Models}:}
\begin{itemize}
  \item PixelRNN
  \item PixelCNN
  \item WaveNet
  \item NADE (Neural Autoregressive Distribution Estimation)
  \item RNADE (Rectified NADE)
  \item MD-RNN (Mixture Density RNN)
  \item DRAW (Deep Recurrent Attentive Writer)
  \item CoNADE (Compositional NADE)
  \item Transformer-based autoregressive models (e.g., Image Transformer)
\end{itemize}

\textcolor{family-hybrid}{\family{family-hybrid}{Hybrid Models}:}
\begin{itemize}
  \item VQ-VAE-2 (Vector Quantized VAE)
  \item GANs with VAEs (e.g., $\beta$-VAE-GAN)
  \item JEM (Joint Energy-Based Model)
\end{itemize}

\textcolor{family-diffusion}{\family{family-diffusion}{Diffusion Models}:}
\begin{itemize}
  \item Noise-Contrastive Estimation (NCE)
  \item Score-Based Generative Models (e.g., PixelSNAIL)
\end{itemize}

\textcolor{family-other}{\family{family-other}{Other notable models}:}
\begin{itemize}
  \item Adversarial Autoencoders
  \item ALI (Adversarially Learned Inference)
  \item Adversarial Variational Bayes (AVB)
  \item GANomaly
  \item Skip-GANomaly
  \item URGANS (Unitary Evolution Recurrent GAN)
  \item VAE-ResNet
  \item OpenAI's CLIP (Contrastive Language-Image Pretraining)
  \item Energy-based Models (EBMs) like PixelSNAIL, PixelDTN, etc.
\end{itemize}
